Alianza API Client

Create a reusable client class with proper error handling:

Java 
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import okhttp3.*;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class AlianzaAPIClient {
    private static final Logger logger = LoggerFactory.getLogger(AlianzaAPIClient.class);
    private static final MediaType JSON = MediaType.get("application/json; charset=utf-8");

    private final OkHttpClient httpClient;
    private final Gson gson;
    private final String baseUrl;
    private String authToken;
    private String partitionId;

    public AlianzaAPIClient(String baseUrl) {
        this.baseUrl = baseUrl.endsWith("/") ? baseUrl.substring(0, baseUrl.length() - 1) : baseUrl;
        this.gson = new Gson();

        this.httpClient = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(30, TimeUnit.SECONDS)
                .writeTimeout(30, TimeUnit.SECONDS)
                .build();
    }

    public AlianzaAPIClient() {
        this("https://api.alianza.com");
    }

    /**
     * Authenticate and obtain auth token
     */
    public AuthResponse authenticate(String username, String password) throws AlianzaAPIException {
        JsonObject credentials = new JsonObject();
        credentials.addProperty("userName", username);
        credentials.addProperty("password", password);

        String responseJson = post("/v2/authorize", credentials.toString());
        AuthResponse authResponse = gson.fromJson(responseJson, AuthResponse.class);

        this.authToken = authResponse.getAuthToken();
        this.partitionId = authResponse.getPartitionId();

        logger.info("Authenticated successfully for partition: {}", partitionId);
        return authResponse;
    }

    /**
     * Make GET request
     */
    public String get(String endpoint) throws AlianzaAPIException {
        Request request = new Request.Builder()
                .url(baseUrl + endpoint)
                .addHeader("Content-Type", "application/json")
                .addHeader("X-AUTH-TOKEN", authToken)
                .get()
                .build();

        return executeRequest(request);
    }

    /**
     * Make POST request
     */
    public String post(String endpoint, String jsonBody) throws AlianzaAPIException {
        RequestBody body = RequestBody.create(jsonBody, JSON);

        Request.Builder requestBuilder = new Request.Builder()
                .url(baseUrl + endpoint)
                .addHeader("Content-Type", "application/json")
                .post(body);

        if (authToken != null) {
            requestBuilder.addHeader("X-AUTH-TOKEN", authToken);
        }

        return executeRequest(requestBuilder.build());
    }

    /**
     * Make PUT request
     */
    public String put(String endpoint, String jsonBody) throws AlianzaAPIException {
        RequestBody body = RequestBody.create(jsonBody, JSON);

        Request request = new Request.Builder()
                .url(baseUrl + endpoint)
                .addHeader("Content-Type", "application/json")
                .addHeader("X-AUTH-TOKEN", authToken)
                .put(body)
                .build();

        return executeRequest(request);
    }

    /**
     * Make DELETE request
     */
    public String delete(String endpoint) throws AlianzaAPIException {
        Request request = new Request.Builder()
                .url(baseUrl + endpoint)
                .addHeader("Content-Type", "application/json")
                .addHeader("X-AUTH-TOKEN", authToken)
                .delete()
                .build();

        return executeRequest(request);
    }

    /**
     * Execute HTTP request
     */
    private String executeRequest(Request request) throws AlianzaAPIException {
        try (Response response = httpClient.newCall(request).execute()) {
            String responseBody = response.body() != null ? response.body().string() : "";

            if (!response.isSuccessful()) {
                logger.error("API request failed: {} {} - Status: {}",
                        request.method(), request.url(), response.code());
                throw new AlianzaAPIException(
                        "API request failed with status: " + response.code(),
                        response.code(),
                        responseBody
                );
            }

            logger.debug("API request succeeded: {} {} - Status: {}",
                    request.method(), request.url(), response.code());
            return responseBody;

        } catch (IOException e) {
            logger.error("Network error during API request: {}", e.getMessage());
            throw new AlianzaAPIException("Network error: " + e.getMessage(), 0, null);
        }
    }

    public String getPartitionId() {
        return partitionId;
    }

    public Gson getGson() {
        return gson;
    }
}

import com.google.gson.reflect.TypeToken;
import java.lang.reflect.Type;

public class ListAccountsExample {

            AlianzaAPIClient client = new AlianzaAPIClient();

            String responseJson = client.get(

            List<Account> accounts = client.getGson().fromJson(responseJson, listType);

Error Handling

Comprehensive error handling with retry logic:

Java 
import java.util.concurrent.TimeUnit;

public class ErrorHandlingExample {
    private static final int MAX_RETRIES = 3;
    private static final int RETRY_DELAY_MS = 1000;

    public static void main(String[] args) {
        AlianzaAPIClient client = new AlianzaAPIClient();

        try {
            // Authenticate
            client.authenticate("username", "password");

            // Execute operation with retry logic
            String result = executeWithRetry(() -> {
                return client.post("/v2/partition/123/account", "{}");
            });

            System.out.println("Success: " + result);

        } catch (AlianzaAPIException e) {
            handleAPIError(e);
        } catch (Exception e) {
            System.err.println("Unexpected error: " + e.getMessage());
            e.printStackTrace();
        }
    }

    /**
     * Execute operation with exponential backoff retry
     */
    private static String executeWithRetry(APIOperation operation)
            throws AlianzaAPIException {
        int retries = 0;
        long delay = RETRY_DELAY_MS;

        while (retries < MAX_RETRIES) {
            try {
                return operation.execute();

            } catch (AlianzaAPIException e) {
                // Don't retry on client errors (4xx)
                if (e.getHttpCode() >= 400 && e.getHttpCode() < 500) {
                    throw e;
                }

                retries++;
                if (retries >= MAX_RETRIES) {
                    throw e;
                }

                System.err.println("Request failed, retrying in " + delay + "ms... " +
                        "(Attempt " + retries + "/" + MAX_RETRIES + ")");

                try {
                    TimeUnit.MILLISECONDS.sleep(delay);
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    throw e;
                }

                delay *= 2; // Exponential backoff
            }
        }

        throw new AlianzaAPIException("Max retries exceeded", 0, null);
    }

    /**
     * Handle API errors with specific status codes
     */
    private static void handleAPIError(AlianzaAPIException e) {
        switch (e.getHttpCode()) {
            case 400:
                System.err.println("Bad Request: Invalid data provided");
                System.err.println("Details: " + e.getResponseBody());
                break;

            case 401:
                System.err.println("Unauthorized: Check your credentials");
                break;

            case 404:
                System.err.println("Not Found: Resource doesn't exist");
                break;

            case 409:
                System.err.println("Conflict: Resource already exists");
                break;

            case 429:
                System.err.println("Rate Limited: Too many requests");
                System.err.println("Please wait before retrying");
                break;

            case 500:
            case 502:
            case 503:
                System.err.println("Server Error: Service temporarily unavailable");
                break;

            default:
                System.err.println("API Error: " + e.getMessage());
        }

        e.printStackTrace();
    }

    @FunctionalInterface
    interface APIOperation {
        String execute() throws AlianzaAPIException;
    }
}

Java 
public class CustomerProvisioningService {
    private final AlianzaAPIClient client;

    public CustomerProvisioningService(String baseUrl) {
        this.client = new AlianzaAPIClient(baseUrl);
    }

    public ProvisioningResult provisionNewCustomer(
            String username,
            String password,
            CustomerData customerData) throws AlianzaAPIException {

        try {
            // Step 1: Authenticate
            System.out.println("1. Authenticating...");
            client.authenticate(username, password);
            String partitionId = client.getPartitionId();
            System.out.println("   ‚úì Authenticated\n");

            // Step 2: Create Account
            System.out.println("2. Creating account...");
            Account account = createAccount(partitionId, customerData);
            System.out.println("   ‚úì Account created: " + account.getAccountId() + "\n");

            // Step 3: Create End User
            System.out.println("3. Creating end user...");
            EndUser user = createEndUser(partitionId, account.getAccountId(), customerData);
            System.out.println("   ‚úì End user created: " + user.getEndUserId() + "\n");

            // Step 4: Create Device
            System.out.println("4. Creating device...");
            Device device = createDevice(partitionId, account.getAccountId(),
                    user.getEndUserId(), customerData);
            System.out.println("   ‚úì Device created: " + device.getDeviceId());
            System.out.println("   ‚úì SIP Credentials:");
            System.out.println("     Username: " + device.getSipUsername());
            System.out.println("     Password: " + device.getSipPassword() + "\n");

            // Step 5: Provision Phone Number
            System.out.println("5. Provisioning phone number...");
            PhoneNumber phone = provisionPhoneNumber(partitionId, account.getAccountId(),
                    user.getEndUserId(), customerData);
            System.out.println("   ‚úì Phone number provisioned: " +
                    phone.getPhoneNumber() + "\n");

            System.out.println("‚úÖ Customer provisioned successfully!\n");

            return new ProvisioningResult(account, user, device, phone);

        } catch (AlianzaAPIException e) {
            System.err.println("‚ùå Provisioning failed: " + e.getMessage());
            throw e;
        }
    }

    private Account createAccount(String partitionId, CustomerData data)
            throws AlianzaAPIException {
        Account account = new Account();
        account.setAccountNumber(data.getAccountNumber());
        account.setName(data.getCompanyName());
        account.setTimeZone(data.getTimeZone());
        account.setStatus("ACTIVE");
        account.setAccountType("BUSINESS");

        String json = client.getGson().toJson(account);
        String response = client.post(
                String.format("/v2/partition/%s/account", partitionId), json);

        return client.getGson().fromJson(response, Account.class);
    }

    private EndUser createEndUser(String partitionId, String accountId,
            CustomerData data) throws AlianzaAPIException {
        EndUser user = new EndUser();
        user.setFirstName(data.getFirstName());
        user.setLastName(data.getLastName());
        user.setEmail(data.getEmail());
        user.setExtension(data.getExtension());
        user.setStatus("ACTIVE");

        String json = client.getGson().toJson(user);
        String response = client.post(
                String.format("/v2/partition/%s/account/%s/enduser",
                        partitionId, accountId), json);

        return client.getGson().fromJson(response, EndUser.class);
    }

    // Additional methods for device and phone number creation...

    public static void main(String[] args) {

                new CustomerProvisioningService("https://api.alianza.com");

        );

                    "api-username",

Best Practices

1. Use Configuration Management

Load credentials from configuration files or environment variables:

// application.properties
alianza.api.url=https://api.alianza.com
alianza.api.username=${ALIANZA_USERNAME}
alianza.api.password=${ALIANZA_PASSWORD}

// Load in Java
Properties props = new Properties();
props.load(new FileInputStream("application.properties"));
String apiUrl = props.getProperty("alianza.api.url");
2. Implement Connection Pooling

OkHttp automatically manages connection pooling for optimal performance.

3. Use Dependency Injection

For Spring applications, use dependency injection:

@Configuration
public class AlianzaConfig {
    @Bean
    public AlianzaAPIClient alianzaClient(
            @Value("${alianza.api.url}") String apiUrl) {
        return new AlianzaAPIClient(apiUrl);
    }
}

@Service
public class AccountService {
    private final AlianzaAPIClient client;

    @Autowired
    public AccountService(AlianzaAPIClient client) {
        this.client = client;
    }
}
